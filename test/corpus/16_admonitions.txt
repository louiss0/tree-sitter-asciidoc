================================================================================
Admonition Blocks with Example
================================================================================

[NOTE]
====
This is a note block with multiple lines.

It can contain paragraphs and other content.
====

[TIP]
====
Pro tip: Always read the documentation first!

- This can include
- Lists and other content
====

--------------------------------------------------------------------------------

(source_file
  (admonition_block
    label: (admonition_block_label)
    content: (example_block
      (example_open)
      (block_content)
      (block_content)
      (block_content)
      (example_close)))
  (admonition_block
    label: (admonition_block_label)
    content: (example_block
      (example_open)
      (block_content)
      (block_content)
      (block_content)
      (block_content)
      (example_close))))

================================================================================
Admonition Blocks with Different Delimited Block Types
================================================================================

[WARNING]
----
# This is dangerous code
rm -rf /
----

[IMPORTANT]
....
This is literal content.
  With preserved formatting.
....

[CAUTION]
****
This is a sidebar admonition.
****

--------------------------------------------------------------------------------

(source_file
  (admonition_block
    label: (admonition_block_label)
    content: (listing_block
      (listing_open)
      (block_content)
      (block_content)
      (listing_close)))
  (admonition_block
    label: (admonition_block_label)
    content: (literal_block
      (literal_open)
      (block_content)
      (block_content)
      (literal_close)))
  (admonition_block
    label: (admonition_block_label)
    content: (sidebar_block
      (sidebar_open)
      (block_content)
      (sidebar_close)))
